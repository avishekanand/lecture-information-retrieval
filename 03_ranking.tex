\section{From Vector Space to Probabilistic Ranking}

\begin{frame}[plain]
\sectionpage
\end{frame}

\begin{frame}{Recap: What We Know}
\begin{block}{Vector Space Model (Lecture 1)}
\begin{itemize}
    \item Documents and queries as vectors in term space
    \item Similarity measured by cosine distance
    \item Foundation for understanding term-based retrieval
\end{itemize}\cpause
\end{block}

\begin{block}{Indexing (Lecture 2)}
\begin{itemize}
    \item Inverted index: term $\to$ posting list (docID, tf, positions)
    \item Accumulator-based scoring: only touch documents with term overlap
    \item Dense indexing: IVF+PQ for semantic vectors
\end{itemize}\cpause
\end{block}

\begin{alertblock}{Today's Question}
We can store and retrieve efficiently. But \textbf{how do we score?}
\end{alertblock}
\end{frame}

\begin{frame}[shrink=10]{The Probability Ranking Principle (PRP)}
\begin{block}{Theorem (Robertson, 1977)}
If documents are ranked by decreasing probability of relevance given the query, then the expected effectiveness of the system is maximized---under the assumption of independent relevance.
\end{block}
\cpause
\vspace{1em}

\textbf{Formally:} Rank by $P(\text{relevant} \mid q, d)$

\vspace{1em}

\textbf{Implications:}
\begin{itemize}
    \item Gives a \textbf{theoretical justification} for probabilistic scoring
    \item All scoring functions (TF-IDF, BM25, LM) are approximations of this ideal
    \item Independence assumption is strong---violated in practice (documents are correlated)
\end{itemize}
\cpause

\vspace{0.5em}

\begin{exampleblock}{Key Insight}
The PRP tells us \emph{what} to optimize. BM25 and Language Models tell us \emph{how}.
\end{exampleblock}
\end{frame}

\begin{frame}[shrink=5]{The Ranking Problem}
\begin{block}{Formal Definition}
\textbf{Given:}
\begin{itemize}
    \item Query $q$ (sequence of terms)
    \item Document collection $D$
    \item Inverted index (from Lecture 2)
\end{itemize}

\vspace{0.5em}
\textbf{Return:} Ranked list scored by relevance $P(\text{relevant}|q,d)$
\end{block}
\cpause
\vspace{1em}

\begin{exampleblock}{Key Insight}
Not all term matches are equal:
\begin{itemize}
    \item Rare terms (``syzygy'') more informative than common (``the'')
    \item Term frequency matters, but with diminishing returns
    \item Document length affects scoring
\end{itemize}\cpause
\end{exampleblock}
\end{frame}

\begin{frame}{Term Weighting: TF-IDF}
\begin{columns}
\begin{column}{0.5\textwidth}
\textbf{Term Frequency (TF)}
\begin{itemize}
    \item More occurrences $\rightarrow$ more important
    \item $\text{tf}(t,d) = \text{count}(t,d)$
\end{itemize}\cpause

\vspace{1em}

\textbf{Inverse Document Frequency (IDF)}
\begin{itemize}
    \item Rare terms $\rightarrow$ more discriminative
    \item $\text{idf}(t) = \log\frac{N}{\text{df}_t}$
\end{itemize}
\end{column}

\begin{column}{0.5\textwidth}
\begin{block}{Combined Score}
$$\text{TF-IDF}(t,d) = \text{tf}(t,d) \times \text{idf}(t)$$
\end{block}

% \vspace{1em}
% \cpause
% \begin{alertblock}{Problem}
% Linear TF growth unrealistic:
% \begin{itemize}
%     \item 100 occurrences $\neq$ 100$\times$ more relevant than 1
%     \item Need saturation
% \end{itemize}
% \end{alertblock}
\end{column}
\end{columns}
\end{frame}

\section{BM25: The Probabilistic Standard}

\begin{frame}[plain]
\sectionpage
\end{frame}

\begin{frame}{Motivation: TF Saturation}
\begin{columns}
\begin{column}{0.5\textwidth}
\textbf{Observation:}
\begin{itemize}
    \item Relevance increases with TF
    \item But saturates quickly
\end{itemize}\cpause

\vspace{1em}

\textbf{Example:}
\begin{itemize}
    \item ``algorithm'' 5$\times$ vs. 50$\times$ 
    \item 5$\rightarrow$10 is meaningful
    \item 45$\rightarrow$50 barely noticeable
\end{itemize}\cpause
\end{column}

\begin{column}{0.5\textwidth}
\input{tikz/tf_saturation_comparison.tex}
\end{column}
\end{columns}

\vspace{1em}

\begin{alertblock}{Need}
Non-linear TF component that saturates gracefully
\end{alertblock}
\end{frame}

\begin{frame}{BM25 Formula}
\begin{block}{Complete Formula}
$$\text{BM25}(q,d) = \sum_{t \in q} \text{IDF}(t) \cdot \frac{f(t,d) \cdot (k_1+1)}{f(t,d) + k_1 \cdot \left(1-b+b \cdot \frac{|d|}{\text{avgdl}}\right)}$$
\end{block}

\vspace{1em}

\textbf{Three components:}
\begin{enumerate}
    \item \textbf{IDF:} Term importance (like TF-IDF)
    \item \textbf{Saturating TF:} Non-linear term frequency
    \item \textbf{Length normalization:} Penalize long documents
\end{enumerate}

\vspace{1em}

\begin{center}
\textbf{Standard parameters: $k_1 = 1.2$, $b = 0.75$}
\end{center}
\end{frame}

\begin{frame}{BM25 Component 1: IDF}
\begin{block}{IDF Formula}
$$\text{IDF}(t) = \log\frac{N - \text{df}_t + 0.5}{\text{df}_t + 0.5}$$
\end{block}

where:
\begin{itemize}
    \item $N$ = total number of documents
    \item $\text{df}_t$ = document frequency of term $t$
\end{itemize}\cpause

\vspace{1em}

\textbf{Properties:}
\begin{itemize}
    \item Similar to TF-IDF but probabilistically motivated
    \item Can be negative for very common terms ($\text{df} > N/2$)
    \item Rare terms get high positive IDF
    \item Common terms get low or negative IDF
\end{itemize}\cpause
\end{frame}

\begin{frame}{BM25 Component 2: Saturating TF}
\begin{block}{Saturating TF Formula}
$$\frac{f(t,d) \cdot (k_1+1)}{f(t,d) + k_1}$$
\end{block}

\textbf{Parameter $k_1$ (typically 1.2--2.0):}
\begin{itemize}
    \item $k_1 \to 0$: Boolean matching (TF doesn't matter)
    \item $k_1 \to \infty$: Linear TF (like TF-IDF)
    \item $k_1 = 1.2$: Moderate saturation (standard)
\end{itemize}\cpause

\vspace{1em}

\begin{columns}
\begin{column}{0.5\textwidth}
\small
\begin{tabular}{ccc}
\toprule
\textbf{TF} & \textbf{$k_1=1.2$} & \textbf{$k_1=2.0$} \\
\midrule
1 & 0.55 & 0.67 \\

\end{tabular}
\end{column}

\begin{column}{0.5\textwidth}
\input{tikz/bm25_k1_comparison.tex}
\end{column}
\end{columns}
\end{frame}

\begin{frame}[shrink=5]{BM25 Component 3: Length Normalization}
\begin{block}{Length Normalization Term}
$$1 - b + b \cdot \frac{|d|}{\text{avgdl}}$$
\end{block}

\textbf{Parameter $b$ (typically 0.75):}
\begin{itemize}
    \item $b = 0$: No length normalization
    \item $b = 1$: Full normalization (penalize long docs heavily)
    \item $b = 0.75$: Partial normalization (standard)
\end{itemize}\cpause

\vspace{1em}

\textbf{Why needed:}
\begin{itemize}
    \item Long documents have more terms just by chance
    \item Should we penalize or not? Depends on collection
    \item $b$ controls the trade-off
\end{itemize}\cpause
\end{frame}

\begin{frame}[shrink=15]{BM25 Example: Effect of Document Length}
\textbf{Scenario:} Term appears 20 times, $k_1=1.2$, $b=0.75$, avgdl=1000

\vspace{0.5em}

\begin{columns}
\begin{column}{0.5\textwidth}
\textbf{Short Document (500 words)}
\begin{align*}
\text{norm} &= 1 - 0.75 + 0.75 \cdot \frac{500}{1000} \\
&= 0.625 \\
\text{TF component} &= \frac{20 \cdot 2.2}{20 + 1.2 \cdot 0.625} \\
&= \frac{44}{20.75} = 2.12
\end{align*}
\end{column}

\begin{column}{0.5\textwidth}
\textbf{Long Document (2000 words)}
\begin{align*}
\text{norm} &= 1 - 0.75 + 0.75 \cdot \frac{2000}{1000} \\
&= 1.75 \\
\text{TF component} &= \frac{20 \cdot 2.2}{20 + 1.2 \cdot 1.75} \\
&= \frac{44}{22.1} = 1.99
\end{align*}
\end{column}
\end{columns}

\vspace{0.5em}

\begin{alertblock}{Observation}
Same TF, but short document scores higher due to length penalty
\end{alertblock}

\section{Statistical Language Models for Ranking}

\begin{frame}[plain]
\sectionpage
\end{frame}

\begin{frame}{A Different Ranking Philosophy}
\begin{columns}
\begin{column}{0.48\textwidth}
\textbf{BM25 Approach:}
\begin{itemize}
    \item ``How well does document match query?''
    \item Scoring based on term overlap
    \item Weighted by importance (IDF, TF)
\end{itemize}\cpause

\vspace{0.5em}

\begin{block}{Matching Score}
% $$\text{score}(q,d) = \sum_{t} w(t,q,d)$$
$$\text{BM25}(q,d) = \sum_{t \in q} \text{IDF}(t) \cdot \frac{f(t,d) \cdot (k_1+1)}{f(t,d) + k_1 \cdot \left(1-b+b \cdot \frac{|d|}{\text{avgdl}}\right)}$$
\end{block}
\end{column}

\begin{column}{0.48\textwidth}
\textbf{Language Model Approach:}
\begin{itemize}
    \item ``How likely would this document generate this query?''
    \item Rank by $P(q|d)$
    \item Probabilistic generative model
\end{itemize}\cpause

\vspace{0.5em}

\begin{block}{Generative Probability}
$$\text{score}(q,d) = P(q|d)$$
\end{block}
\end{column}
\end{columns}

\vspace{1em}

\begin{exampleblock}{Key Shift}
BM25 asks \textit{``does this document contain the right words?''} \\
Language Models ask \textit{``would a reader of this document ever write this query?''}
\end{exampleblock}
\cpause
\end{frame}

\begin{frame}{The Generative Story}
\begin{exampleblock}{Intuition}
Imagine randomly sampling words from a document. The document that is \textbf{most likely to produce your query terms} is ranked highest.
\end{exampleblock}

\vspace{1em}

\textbf{How it works step by step:}
\begin{enumerate}
    \item Pick a document $d$ from the collection\cpause
    \item The document defines a \textbf{language model} --- a probability distribution $P(w|d)$ over words\cpause
    \item \textbf{Sample} query words from this distribution: $P(q|d) = \prod_{w \in q} P(w|d)$\cpause
    \item Rank documents by $P(q|d)$ --- those more likely to ``emit'' the query rank higher\cpause
\end{enumerate}

\vspace{0.5em}

% \begin{alertblock}{Practical consequence}
% A document that uses the query words frequently (and exclusively) will get a high score. But a document missing even one query word gets $P(q|d) = 0$ --- this is why smoothing matters.
% \end{alertblock}
\end{frame}

\begin{frame}[shrink=12]{Query Likelihood Model: Philosophy}
\begin{block}{The Generative Story}
Instead of matching keywords, we ask: \textbf{``What is the probability that this document would produce this query?''}
\end{block}
\cpause
\vspace{1em}

\textbf{Mathematical Formulation:}
\begin{itemize}
    \item Assume query terms $w \in q$ are sampled \textbf{independently} from the document $d$ (unigram assumption).
    \cpause
    \item Rank documents by:
    $$P(q|d) = \prod_{w \in q} P(w|d)$$
    \cpause
    \item In practice, we use log-probabilities to avoid underflow:
    $$\text{score}(q,d) = \sum_{w \in q} \log P(w|d)$$
\end{itemize}\cpause
\end{frame}

\begin{frame}[shrink=12]{Estimation}
\begin{block}{Maximum Likelihood Estimate (MLE)}
How do we calculate $P(w|d)$? The simplest way is term frequency:
$$P_{\text{ML}}(w|d) = \frac{\text{count}(w,d)}{|d|}$$
\end{block}

But what is the catch here?
\cpause
\vspace{0.5em}

\begin{alertblock}{The Catch}
If a query term $w$ is \textbf{not in the document}, then $P_{\text{ML}}(w|d) = 0$.
\begin{itemize}
    \item The product $P(q|d) = 0$ for the entire query!
    \item One missing term kills the score, even for a 90\% relevant document.
\end{itemize}\cpause
\end{alertblock}

\vspace{0.5em}

\textbf{Example:} Query ``machine learning algorithms'' misses a document about ``machine learning'' because the word ``algorithms'' is absent.
\end{frame}

\begin{frame}{Smoothing: Core Idea}
\begin{block}{Principle}
Mix document evidence with collection evidence:
$$P(w|d) = f(\text{document evidence}, \text{collection evidence})$$
\end{block}

\vspace{1em}

\textbf{Collection Language Model:}
$$P(w|C) = \frac{\sum_{d' \in C} \text{count}(w,d')}{\sum_{d' \in C} |d'|}$$

\vspace{1em}

% \textbf{Two Main Approaches:}
% \begin{enumerate}
%     \item \textbf{Jelinek-Mercer:} Fixed linear interpolation
%     \item \textbf{Dirichlet:} Length-adaptive interpolation
% \end{enumerate}

% \vspace{0.5em}

\begin{exampleblock}{Effect}
Even if $\text{count}(w,d) = 0$, we have $P(w|d) > 0$ from collection
\end{exampleblock}
\end{frame}

\begin{frame}[shrink=5]{Jelinek-Mercer Smoothing}
\begin{block}{Formula}
$$P(w|d) = (1-\lambda) \cdot P_{\text{ML}}(w|d) + \lambda \cdot P(w|C)$$
\end{block}

\textbf{Parameter $\lambda$ (typically 0.1--0.3):}
\begin{itemize}
    \item $\lambda = 0$: Only use document (no smoothing)
    \item $\lambda = 1$: Only use collection (ignore document)
    \item $\lambda = 0.1$: Mostly trust document, small collection backup
\end{itemize}\cpause

\vspace{1em}
\cpause
\textbf{Properties:}
\begin{itemize}
    \item \textcolor{green}{\checkmark} Simple and interpretable
    \item \textcolor{green}{\checkmark} Same $\lambda$ for all documents
    \item \textcolor{red}{\ding{55}} Doesn't adapt to document length
\end{itemize}
\end{frame}

\begin{frame}{Dirichlet Smoothing}
\begin{block}{Formula}
$$P(w|d) = \frac{\text{count}(w,d) + \mu \cdot P(w|C)}{|d| + \mu}$$
\end{block}

\textbf{Parameter $\mu$ (typically 1000--2500):}
\begin{itemize}
    \item Smoothing strength (number of ``pseudo-counts'' from collection)
    \item Higher $\mu$: More smoothing. Lower $\mu$: Less smoothing.
\end{itemize}\cpause

\vspace{1em}

\textbf{Key Difference: Length-Adaptive}
\begin{itemize}
    \item \textbf{Short documents:} $\mu$ dominates denominator $\rightarrow$ rely more on collection
    \item \textbf{Long documents:} Actual counts dominate $\rightarrow$ rely on document
    \item Automatically adapts smoothing to document length!
\end{itemize}\cpause
\end{frame}

\begin{frame}{Query Likelihood Scoring: Step by Step}
\begin{exampleblock}{The Recipe}
Given query $q$ and document $d$, score = $\log P(q|d)$:
\end{exampleblock}

\vspace{0.4em}

\begin{enumerate}
    \item \textbf{Estimate} via MLE: $\displaystyle P_{\text{ML}}(w|d) = \frac{\text{count}(w,d)}{|d|}$\cpause
    \cpause
    \vspace{0.2em}
    \item \textbf{Smooth} (JM): $\hat{P}(w|d) = (1-\lambda)\,P_{\text{ML}}(w|d) + \lambda\,P(w|C)$\cpause
    \cpause

    \item \textbf{Score}: $\displaystyle\text{score}(q,d) = \sum_{w \in q} \log \hat{P}(w|d)$\cpause
    \cpause
    \vspace{0.2em}
    \item \textbf{Rank} by score --- higher score = more relevant.\cpause
\end{enumerate}

\begin{frame}{RM3: Relevance Model 3}
\begin{block}{Foundation}
Provides a probabilistic way to perform query expansion by building a \textbf{model of relevance} ($R$) from the initial top-$k$ results.
\end{block}

\vspace{0.5em}

\textbf{Phase 1: Estimating the Relevance Model $P(w|R)$}
We want to find terms $w$ that are likely to appear in relevant documents:
$$P(w|R) \approx \sum_{d \in \text{top-}k} \underbrace{P(w|d)}_{\text{term presence}} \cdot \underbrace{P(d|q)}_{\text{doc relevance weight}}$$

\begin{itemize}
    \item $P(w|d)$ is the smoothed language model of document $d$.
    \item $P(d|q)$ is the score of the document from the first retrieval stage.
    \item Terms common across many high-scoring documents get the highest $P(w|R)$.
\end{itemize}\cpause
\end{frame}

\begin{frame}{RM3: Interpolation and Final Query}
\textbf{Phase 2: Combining with the Original Query}
To prevent the query from drifting too far, we interpolate the relevance model with the original query's word distribution:

\begin{tcolorbox}[colback=blue!5, colframe=blue!50, title=RM3 Linear Interpolation]
$$P(w|q_{\text{new}}) = (1-\lambda) P(w|q_{\text{orig}}) + \lambda P(w|R)$$
\end{tcolorbox}

\begin{itemize}
    \item $\lambda=1$: Full query expansion (total trust in top-$k$).
    \item $\lambda=0$: No expansion (original query only).
    \item \textbf{Common default:} $\lambda \approx 0.5$.
\end{itemize}\cpause

\vspace{0.5em}

\begin{exampleblock}{The Final Retrieval}
The top \texttt{fb\_terms} from $P(w|q_{\text{new}})$ are used to form a new weighted query for the second stage.
\end{exampleblock}
\end{frame}

\begin{frame}[shrink=12]{RM3: Key Parameters}
\begin{block}{fb\_docs (Feedback Documents): 3--10}
How many top documents to use for expansion
\begin{itemize}
    \item Too few: May miss relevant terms. Too many: Risk non-relevant docs.
    \item \textbf{Standard:} 10
\end{itemize}\cpause
\end{block}

\begin{block}{fb\_terms (Feedback Terms): 10--20}
How many expansion terms to add
\begin{itemize}
    \item Too few: Limited benefit. Too many: Noise.
    \item \textbf{Standard:} 10
\end{itemize}\cpause
\end{block}

\begin{block}{$\lambda$ (Interpolation Weight): 0.5--0.8}
Original vs. expansion weight
\begin{itemize}
    \item Lower (0.5): Trust original query more
    \item Higher (0.8): Trust expansion more
\end{itemize}\cpause
\end{block}
\end{frame}

\begin{frame}[shrink=30]{Strategy 3: Chain-of-Thought Query Decomposition --- Method}
\textbf{Prompt:}
\begin{tcolorbox}[colback=gray!10, colframe=gray!50, fontupper=\small\ttfamily]
I need to answer: "How did the 2008 financial crisis affect European housing markets?"\\
Break this into 3-4 simpler search queries that together would cover the full answer.
\end{tcolorbox}

\vspace{0.3em}

\textbf{LLM Output:}
\begin{enumerate}
    \item ``2008 financial crisis causes subprime mortgage''
    \item ``European housing prices 2008 2009 decline''
    \item ``EU bank bailouts housing market impact''
    \item ``European mortgage lending regulations post-crisis''
\end{enumerate}\cpause

\vspace{0.3em}

\begin{block}{Pipeline}
\texttt{BM25$_1$ $\parallel$ BM25$_2$ $\parallel$ BM25$_3$ $\parallel$ BM25$_4$} $\to$ merge (union / RRF) $\to$ better recall than single complex query
\end{block}

\vspace{0.3em}

\begin{exampleblock}{Best for}
Multi-hop reasoning queries, complex informational needs. \textbf{Not} for simple keyword lookups.
\end{exampleblock}
\end{frame}

\begin{frame}{Strategy 4: Query2Doc --- Intuition}
\begin{exampleblock}{The Problem with Expansion Terms}
Adding isolated keywords (``neural'', ``deep'', ``algorithm'') helps, but loses the context of how those words relate to each other. BM25 treats each term independently.
\end{exampleblock}

\vspace{0.5em}

\begin{block}{Core Idea (Wang et al., 2023)}
Generate a full \textbf{pseudo-document} with an LLM and \emph{prepend it to the original query}. BM25 now sees a rich, contextual query that looks like the documents it is searching over.
\end{block}

\vspace{0.5em}

\end{frame}

\begin{frame}[shrink=20]{Strategy 4: Query2Doc --- Method}
\textbf{Original query:} ``dark matter detection methods''

\vspace{0.3em}

\textbf{LLM-generated pseudo-doc:}
\begin{quote}
\small\itshape
``Dark matter detection experiments include direct detection using cryogenic detectors, indirect detection through gamma-ray observations, and collider searches at the Large Hadron Collider (LHC). WIMP-based detectors and axion searches are among the leading approaches...''
\end{quote}

\textbf{Expanded query to BM25:} \texttt{[original query] + [pseudo-doc text]}

\cpause

\vspace{0.3em}

\begin{columns}[T]
\begin{column}{0.48\textwidth}
\begin{block}{\textcolor{irHybrid}{\checkmark} Advantages}
\begin{itemize}\small
    \item Drops into \textit{any} BM25 system unchanged
    \item No prior retrieval step needed
    \item LLM runs only once per query
    \item Rich term co-occurrence context preserved
\end{itemize}
\end{block}
\end{column}
\begin{column}{0.48\textwidth}
\begin{alertblock}{\ding{55} Risks}
\begin{itemize}\small
    \item LLM hallucinations add noise terms
    \item Longer concatenated query = slower scoring
    \item May need weighting: original vs.\ generated
\end{itemize}
\end{alertblock}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Strategy 5: GenQR (Generative Query Reformulation) --- Intuition}
\begin{exampleblock}{The Problem with ``term lists'' expansions}
Synonyms/HyDE/Query2Doc can help, but often produce \emph{bag-of-terms} or long pseudo-text.\\
Sometimes you just want a \textbf{better query} that expresses the same intent.
\end{exampleblock}

\vspace{0.5em}

\begin{block}{Core Idea (Wang et al., 2023)}
Use an LLM to \textbf{rewrite the query itself} into 3--5 high-quality paraphrases, then retrieve using the reformulated query (often appended/interpolated with the original).
\end{block}

\vspace{0.8em}

\begin{center}
\begin{tikzpicture}[node distance=1.1cm, every node/.style={font=\small\sffamily}]
    \node[infra, text width=2.4cm, align=center] (q) {Original\\Query $q_0$};
    \node[dense, right=1.3cm of q] (llm) {GenQR\\(T5 / FLAN-T5)};
    \node[irBoxBase, fill=irHybrid!10, draw=irHybrid, right=1.3cm of llm, text width=3.0cm, align=center] (qr) {Reformulated\\Query $q_r$\\(paraphrases)};
    \node[sparse, right=1.3cm of qr] (bm25) {BM25};
    \node[rerank, right=1.3cm of bm25] (rerank) {Neural\\Re-ranker};

    \draw[irArrowBase] (q) -- (llm);
    \draw[irArrowBase] (llm) -- (qr);
    \draw[irArrowBase] (qr) -- (bm25);
    \draw[irArrowBase] (bm25) -- (rerank);
\end{tikzpicture}
\end{center}

\vspace{0.3em}
\begin{alertblock}{Key point vs PRF}
No dependence on top-$k$ pseudo-relevant docs: GenQR is \textbf{direct reformulation} using only $q_0$.
\end{alertblock}
\end{frame}

\begin{frame}[shrink=22]{Strategy 5: GenQR --- Method (Prompting \& Fine-tuning)}
\textbf{Two common instantiations:}

\vspace{0.3em}
\begin{block}{(A) Prompting an instruction-tuned model (e.g., FLAN-T5)}
\begin{tcolorbox}[colback=gray!10, colframe=gray!50, fontupper=\small\ttfamily]
Improve the search effectiveness by suggesting rewrites for the query:\\
\hspace*{1em} "dark matter detection methods"\\
Return 5 rewritten queries (one per line).
\end{tcolorbox}
\end{block}

\cpause
\vspace{0.4em}

\begin{block}{(B) Fine-tuning a text-to-text model (e.g., T5) for ``refine''}
Train a model to generate $q_r$ given $q_0$ using weak supervision:
\begin{itemize}\small
    \item Build query pairs $\langle q_x, q_y\rangle$ that share relevant documents (same intent)
    \item Filter pairs to reduce drift / noise (e.g., overlap + effectiveness + stopword filtering)
    \item Fine-tune with an explicit task token: \texttt{T5("refine", $q_0$, "</s>")}
\end{itemize}
\end{block}

\vspace{0.3em}
\begin{exampleblock}{Output form}
GenQR outputs \textbf{well-formed rewritten queries} (not just extra keywords), which you can:
\textbf{append} to the original or \textbf{interpolate} (weighted) for first-stage retrieval.
\end{exampleblock}
\end{frame}

\begin{frame}[shrink=22]{Strategy 5: GenQR --- Pipeline Details, When It Works, Risks}
\begin{block}{Retrieval pipeline integration (retrieve-then-rerank)}
\small
\begin{itemize}
    \item Generate $N$ paraphrases (often $N\!=\!5$) using decoding (e.g., beam search), then keep top-$N$.
    \item Form final first-stage query:
    \[
      q^{\dagger} = q_0 \;+\; \beta \cdot q_r
      \quad \text{(append / weight reformulations vs original)}
    \]
    \item Run BM25 on $q^{\dagger}$, then re-rank with a neural model (monoT5 / cross-encoder).
\end{itemize}
\end{block}

\cpause
\vspace{0.4em}

\begin{columns}[T]
\begin{column}{0.49\textwidth}
\begin{block}{\textcolor{irHybrid}{\checkmark} When it works best}
\begin{itemize}\small
    \item Lexical mismatch (multiple valid phrasings)
    \item Natural-language / verbose queries
    \item Collections where BM25 recall is the bottleneck
\end{itemize}
\end{block}
\end{column}
\begin{column}{0.49\textwidth}
\begin{alertblock}{\ding{55} Risks / gotchas}
\begin{itemize}\small
    \item \textbf{Topical drift}: rewrite changes intent (silent failure)
    \item \textbf{Extra latency}: generation per query (mitigate with small models / caching)
    \item Needs careful $\beta$ (too high $\Rightarrow$ rewrite dominates)
\end{itemize}
\end{alertblock}
\end{column}
\end{columns}

\vspace{0.2em}
\begin{exampleblock}{Practical advice}
Treat GenQR as a \textbf{recall booster}. Keep the original query strong; set $\beta$ small and validate.
\end{exampleblock}
\end{frame}




\begin{frame}{Comparison: Classical vs. LLM Query Expansion}
\begin{table}
\centering
\small
\begin{tabular}{lll}
\toprule
\textbf{Aspect} & \textbf{Classical (RM3)} & \textbf{LLM-Based} \\
\midrule
Knowledge source & Top-$k$ retrieved docs & Parametric (training data) \\
Failure mode & Query drift (bad top-$k$) & Hallucination \\
Latency & Fast (stats only) & Slower (LLM inference) \\
Cost & Free & API cost / GPU \\
Domain knowledge & Needs in-domain corpus & Pre-trained, zero-shot \\
Cold-start & Fails (empty index) & Works (no index needed) \\
Interpretability & Terms from real docs & Generated terms \\
\bottomrule
\end{tabular}
\end{table}

\vspace{1em}

\begin{exampleblock}{Best Practice (2024+)}
\textbf{Combine both:} Use LLM expansion to improve the initial retrieval, then apply RM3 on the (now better) top-$k$ results. The two methods are complementary.
\end{exampleblock}
\end{frame}

\begin{frame}{When Query Expansion Helps vs. Hurts}
\begin{columns}
\begin{column}{0.48\textwidth}
\begin{block}{\checkmark Helps When:}
\begin{itemize}
    \item Initial results mostly relevant
    \item Vocabulary mismatch present
    \item Broad informational queries
    \item Query: ``climate change''
    \item Expansion adds: ``warming'', ``CO2'', ``emissions''
\end{itemize}\cpause
\end{block}
\end{column}

\begin{column}{0.48\textwidth}
\begin{alertblock}{\ding{55} Hurts When:}
\begin{itemize}
    \item Initial results off-topic
    \item Ambiguous queries
    \item Very specific queries
    \item Query: ``jaguar''
    \item Expansion adds: ``car'', ``speed'' (wrong sense!)
\end{itemize}\cpause
\end{alertblock}
\end{column}
\end{columns}

\vspace{1em}

\end{frame}

\begin{frame}[shrink=10]{Appendix: Detailed BM25 Worked Example}
\textbf{Query:} ``quantum algorithm'' \quad $N = 100{,}000$, avgdl $= 500$

\vspace{0.5em}

\begin{columns}
\begin{column}{0.45\textwidth}
\textbf{Candidate Doc A:}
\begin{itemize}
    \item $|d| = 300$
    \item tf(quantum) = 5
    \item tf(algorithm) = 3
\end{itemize}\cpause
\end{column}
\begin{column}{0.45\textwidth}
\textbf{Candidate Doc B:}
\begin{itemize}
    \item $|d| = 800$
    \item tf(quantum) = 8
    \item tf(algorithm) = 1
\end{itemize}\cpause
\end{column}
\end{columns}

\vspace{1em}

\textbf{Stats:} df(quantum) = 200, df(algorithm) = 5,000

\vspace{0.5em}

\textbf{1. IDF Calculation:}
\begin{align*}
\text{IDF(q)} &= \log\frac{10^5 - 200 + 0.5}{200 + 0.5} \approx \mathbf{6.21} \\
\text{IDF(a)} &= \log\frac{10^5 - 5000 + 0.5}{5000 + 0.5} \approx \mathbf{2.94}
\end{align*}
\end{frame}

\begin{frame}[shrink=10]{Appendix: BM25 Worked Example (cont.)}
\textbf{2. Final Scores ($k_1=1.2, b=0.75$):}

\vspace{1em}

\begin{table}
\centering
\small
\begin{tabular}{lll}
\toprule
\textbf{Step} & \textbf{Doc A ($|d|=300$)} & \textbf{Doc B ($|d|=800$)} \\
\midrule

\midrule

\midrule
