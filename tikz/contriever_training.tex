% Contriever: Document Views + MoCo Queue Visualization
% Shows: document → random crop → two views → encoder → contrastive loss with queue
\begin{tikzpicture}[
    scale=0.72, transform shape,
    box/.style={draw, rounded corners=3pt, minimum height=0.7cm, align=center, font=\small, drop shadow={opacity=0.1}},
    arr/.style={-{Stealth[length=2.5mm]}, thick},
    crop/.style={draw, dashed, rounded corners=2pt, minimum height=0.55cm, font=\tiny, align=center},
]

% ===== LEFT COLUMN (x=0): Pair Generation =====

\node[box, fill=irInfra!8, draw=irInfra!60, minimum width=3.3cm, minimum height=1.6cm, font=\tiny, text width=2.9cm] (doc) at (0, 0)
    {\textbf{Document $d$}\\[2pt]
     ``Aspirin is a medication used to reduce pain, fever, or inflammation...''};
\node[font=\tiny\sffamily, text=irInfra, above=0.08cm of doc] {Unlabeled corpus};

% Crop arrows
\draw[arr, draw=irSparse!80] ([xshift=-0.5cm]doc.south) -- ++(-0.7, -0.7)
    node[midway, left, font=\tiny\sffamily, text=irSparse!80] {crop 1};
\draw[arr, draw=irDense!80] ([xshift=0.5cm]doc.south) -- ++(0.7, -0.7)
    node[midway, right, font=\tiny\sffamily, text=irDense!80] {crop 2};

% Views
\node[crop, fill=irSparse!12, draw=irSparse, minimum width=2.3cm] (v1) at (-1.5, -2.2)
    {View $v_1$\\``reduce pain, fever''};
\node[crop, fill=irDense!12, draw=irDense, minimum width=2.3cm] (v2) at (1.5, -2.2)
    {View $v_2$\\``treat inflammation''};

% Encoder
\draw[arr, draw=irSparse!80] (v1.south) -- ++(0.7, -0.8);
\draw[arr, draw=irDense!80] (v2.south) -- ++(-0.7, -0.8);

\node[box, fill=irHybrid!15, draw=irHybrid, minimum width=3.8cm, minimum height=0.8cm] (enc) at (0, -4.2)
    {shared Bi-Encoder $f_\theta$};

% Embeddings
\draw[arr, draw=irSparse!80] (-1.2, -4.6) -- (-1.2, -5.2) node[midway, left, font=\tiny] {};
\draw[arr, draw=irDense!80] (1.2, -4.6) -- (1.2, -5.2) node[midway, right, font=\tiny] {};

\node[box, fill=irSparse!20, draw=irSparse, minimum width=1.4cm, font=\footnotesize] (z1) at (-1.2, -5.7) {$\mathbf{z}_1$};
\node[box, fill=irDense!20, draw=irDense, minimum width=1.4cm, font=\footnotesize] (z2) at (1.2, -5.7) {$\mathbf{z}_2$};

% Pull together
\draw[irHybrid!80!black, ultra thick, {Stealth[length=2.2mm]}-{Stealth[length=2.2mm]}]
    (z1.east) -- (z2.west)
    node[midway, above=2pt, font=\tiny\bfseries, text=irHybrid!80!black] {positives};

% ===== RIGHT COLUMN (x=10): Negative Sources =====

% Batch
\node[font=\small\bfseries, text=irInfra] at (10, 0.5) {Current Training Batch};
\foreach \i/\label/\type in {1/$d_1$/neg, 2/$d_2$/neg, 3/$d_i$/pos, 4/$d_B$/neg}{
    \pgfmathsetmacro{\xpos}{8.5 + (\i-1)*1.0}
    \ifnum\i=3
        \node[box, fill=irHybrid!20, draw=irHybrid, minimum width=0.9cm, minimum height=0.45cm, font=\tiny] (b\i) at (\xpos, -0.2) {\label};
        \node[font=\tiny\bfseries, text=irHybrid!80!black] at (\xpos, -0.7) {\type};
    \else
        \node[box, fill=irSparse!10, draw=irSparse!60, minimum width=0.9cm, minimum height=0.45cm, font=\tiny] (b\i) at (\xpos, -0.2) {\label};
        \node[font=\tiny, text=irInfra] at (\xpos, -0.7) {\type};
    \fi
}
\draw[decorate, decoration={brace, amplitude=3pt, mirror}, irInfra]
    (8.05, -0.85) -- (11.95, -0.85)
    node[midway, below=4pt, font=\tiny\sffamily, text=irInfra] {$B-1$ in-batch negatives};

% Queue
\node[font=\small\bfseries, text=irRerank!80!black] at (10, -2.7) {MoCo-style Queue};
\foreach \i in {0,...,7}{
    \pgfmathsetmacro{\xpos}{8.25 + \i*0.5}
    \node[draw=irRerank!50, fill=irRerank!8, rounded corners=1pt, minimum width=0.4cm, minimum height=0.35cm, font=\tiny] (q\i) at (\xpos, -3.4) {$\mathbf{z}$};
}
\node[font=\tiny, text=irInfra] at (12.1, -3.4) {$\cdots$};
\draw[decorate, decoration={brace, amplitude=3pt, mirror}, irRerank!70]
    (8.05, -3.7) -- (12.1, -3.7)
    node[midway, below=4pt, font=\tiny\sffamily, text=irRerank!70] {$K$ past embeddings ($K \gg B$)};

% Enqueue
\draw[arr, draw=irDense!70, dashed, -{Stealth[length=2mm]}] (z2.east) .. controls (4.5, -5.7) and (7.5, -4.5) .. (8.2, -3.5)
    node[pos=0.5, above, font=\tiny\sffamily, text=irDense!70, rotate=15] {enqueue};

% ===== BOTTOM: Loss Function (Clear of overlaps) =====

\node[box, fill=yellow!15, draw=yellow!60!black, minimum width=12cm, minimum height=1.2cm, font=\small] (loss) at (5, -8.5)
    {$\mathcal{L} = -\log \dfrac{\exp(\mathbf{z}_1^\top \mathbf{z}_2 / \tau)}{\exp(\mathbf{z}_1^\top \mathbf{z}_2 / \tau) + \displaystyle\sum_{j \in \text{Batch}} \exp(\mathbf{z}_1^\top \mathbf{z}_j / \tau) + \sum_{k \in \text{Queue}} \exp(\mathbf{z}_1^\top \mathbf{z}_k / \tau)}$};

% Component labels (no arrows for negatives as requested)
\pgfmathsetmacro{\lossTop}{-7.2}

% Positive contribution - keeping this as it defines the pair being trained on
\draw[arr, draw=irHybrid!80!black, shorten >=10pt] (0, -5.6) -- (0, \lossTop);
\node[font=\tiny\bfseries, text=irHybrid!80!black, anchor=west, inner sep=2pt] at (0.05, -6.4) {positives ($z_1, z_2$)};

% Batch contribution label
\node[font=\tiny\sffamily, text=irInfra, anchor=west, inner sep=2pt] at (9.55, -6.4) {batch negatives};

% Queue contribution label
\node[font=\tiny\sffamily, text=irRerank!70, anchor=west, inner sep=2pt] at (11.7, -4.8) {queue negatives};

% Shared embeddings arrows - also keeping for z1, z2 flow
\draw[arr, draw=irSparse!70, shorten >=10pt] (z1.south) -- (-1.1, \lossTop);
\draw[arr, draw=irDense!70, shorten >=10pt] (z2.south) -- (1.1, \lossTop);

\end{tikzpicture}
